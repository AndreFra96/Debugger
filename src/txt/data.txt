1|ordini senza items|SELECT t_order.order_id,t_order.order_date,t_order.id_customer,t_order.id_locale,is_renew,monthly_id FROM t_order LEFT JOIN t_order_item ON t_order.order_id = t_order_item.order_id WHERE t_order_item.order_id IS NULL
2|items senza ordine|SELECT t_order_item.item_id,t_order_item.article_id_trace,item_quantity,item_price FROM t_order_item LEFT JOIN t_order ON t_order_item.order_id = t_order.order_id WHERE t_order.order_id IS NULL
3|rinnovi senza items|SELECT t_rinnovi.rinnovo_id,t_rinnovi.rinnovo_customer_id,t_rinnovi.rinnovo_location_id,t_rinnovi.rinnovo_month_reference,t_rinnovi.previous_rinnovo_id FROM t_rinnovi LEFT JOIN t_rinnovi_items ON t_rinnovi.rinnovo_id = t_rinnovi_items.rinnovo_id WHERE t_rinnovi_items.rinnovo_id IS NULL
4|items senza rinnovo|SELECT t_rinnovi_items.id,t_rinnovi_items.article_id_trace,t_rinnovi_items.price,t_rinnovi_items.serial_number,t_rinnovi_items.date_rinnovo,t_rinnovi_items.t_trace_id,t_rinnovi_items.customer_id,t_rinnovi_items.location_id,t_rinnovi_items.trace_id FROM t_rinnovi_items LEFT JOIN t_rinnovi ON t_rinnovi_items.rinnovo_id = t_rinnovi.rinnovo_id WHERE t_rinnovi.rinnovo_id IS NULL
5|mensili senza items|SELECT t_monthly.monthly_id,t_monthly.monthly_customer_id,t_monthly.monthly_location_id,t_monthly.monthly_next_billing,t_monthly.monthly_type_billing,t_monthly.monthly_payment_id,t_monthly.group_billing FROM t_monthly LEFT JOIN t_monthly_items ON t_monthly.monthly_id = t_monthly_items.monthly_id WHERE t_monthly_items.monthly_id IS NULL
